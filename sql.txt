CREATE TABLE cadastro_pilotos ( user_id BIGINT NOT NULL AUTO_INCREMENT, nome VARCHAR(45) NULL, cpf VARCHAR(45) NULL, rg VARCHAR(45) NULL, data_nascimento VARCHAR(45) NULL,
                        telefone_contato VARCHAR(45) NULL, email VARCHAR(45) NULL, contato_emergencia VARCHAR(45) NULL, PRIMARY KEY (user_id));

DELIMITER // CREATE PROCEDURE sp_createUser( IN p_nome VARCHAR(20), IN p_cpf VARCHAR(20), IN p_data_nascimento VARCHAR(20), IN p_telefone_contato VARCHAR(20), IN p_email VARCHAR(20), IN p_contato_emergencia  VARCHAR(20)) BEGIN IF ( select exists (select 1 from cadastro_pilotos where cpf = p_cpf) ) THEN
select 'CPF Exists !!';
ELSE
insert into cadastro_pilotos (nome, cpf, rg, data_nascimento, telefone_contato, email, contato_emergencia) values (p_nome, p_cpf, p_rg, p_data_nascimento, p_telefone_contato, p_email, p_contato_emergencia);
END IF; END // DELIMITER ;
